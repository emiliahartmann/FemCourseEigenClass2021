66 void IntRule1d::gauleg(const double x1, const double x2, VecDouble &co, VecDouble &w){
67  int n = w.size();
68 
69  double EPS = 1.0e-14;
70  int m, j, i;
71  double z1, z, xm, xl, pp, p3, p2, p1;
72 
73  m = (n + 1) / 2;
74  xm = 0.5 * (x2 + x1);
75  xl = 0.5 * (x2 - x1);
76  for (i = 0; i < m; i++) {
77  z = cos(PI * (i + 0.75) / (n + 0.5));
78  do {
79  p1 = 1.0;
80  p2 = 0.0;
81  for (j = 0; j < n; j++) {
82  p3 = p2;
83  p2 = p1;
84  p1 = ((2.0 * j + 1.0) * z * p2 - j * p3) / (j + 1);
85  }
86  pp = n * (z * p1 - p2) / (z * z - 1.0);
87  z1 = z;
88  z = z1 - p1 / pp;
89  } while (fabs(z - z1) > EPS);
90  co[i] = xm - xl*z;
91  co[n - 1 - i] = xm + xl*z;
92  w[i] = 2.0 * xl / ((1.0 - z * z) * pp * pp);
93  w[n - 1 - i] = w[i];
94  }
