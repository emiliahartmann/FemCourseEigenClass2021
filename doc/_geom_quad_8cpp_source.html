23 void GeomQuad::Shape(const VecDouble &xi, VecDouble &phi, MatrixDouble &dphi) {
24  if(xi.size() <=0 || xi.size() > Dimension) DebugStop();

30  double csi = xi[0];
31  double eta = xi[1];
